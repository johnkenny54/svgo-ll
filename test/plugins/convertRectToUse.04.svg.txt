===

      <rect x="0" y="0" width="10" height="10" fill="gray"/>
      <rect x="10" y="10" width="10" height="10" fill="gray"/>
    </pattern>
  </defs>
  <rect x="5" y="5" width="100" height="100" fill="url(#pattern)" stroke="black" stroke-width="4"/>
  <rect x="120" y="10" width="100" height="100" fill="blue" stroke="black" stroke-width="4"/>
