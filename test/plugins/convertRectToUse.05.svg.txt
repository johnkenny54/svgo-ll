Don't convert common <rect> to <use> if it has mask, filter, or clip.

===

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 300 300">
  <defs>
    <filter id="a" filterUnits="objectBoundingBox" x="0%" y="0%" width="100%" height="100%">
      <feTurbulence type="turbulence" baseFrequency="0.05" numOctaves="2"/>
    </filter>
  </defs>
  <rect x="5" y="5" width="100" height="100" fill="blue"/>
  <rect x="120" y="10" width="100" height="100" fill="blue"/>
  <rect x="10" y="120" width="100" height="100" fill="blue"/>
  <rect x="120" y="120" width="100" height="100" fill="blue" filter="url(#a)"/>
</svg>
