Convert common <rect> to <use>.

===

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 40 40">
  <rect width="1.23456" height="2.34567"/>
  <rect x="2" y="3" width="1.23456" height="2.34567"/>
  <rect x="4" y="6" width="1.23456" height="2.34567"/>
  <rect x="6" y="9" width="1.23456" height="2.34567"/>
</svg>

@@@

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 40 40">
  <use href="#a"/>
  <use x="2" y="3" href="#a"/>
  <use x="4" y="6" href="#a"/>
  <use x="6" y="9" href="#a"/>
  <defs>
    <rect id="a" width="1.23456" height="2.34567"/>
  </defs>
</svg>
