Don't convert common <rect> to <use> if it has a fill with gradientUnits="userSpaceOnUse".

===

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 15 15">
  <defs>
    <linearGradient id="a" gradientUnits="userSpaceOnUse" x1="6" y1="9" x2="7" y2="11">
      <stop offset="0" stop-color="white"/>
      <stop offset="1" stop-color="black"/>
    </linearGradient>
  </defs>
  <rect width="1.23456" height="2.34567"/>
  <rect x="2" y="3" width="1.23456" height="2.34567"/>
  <rect x="4" y="6" width="1.23456" height="2.34567"/>
  <rect x="6" y="9" width="1.23456" height="2.34567" fill="url(#a)"/>
</svg>

@@@

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 15 15">
  <defs>
    <linearGradient id="a" gradientUnits="userSpaceOnUse" x1="6" y1="9" x2="7" y2="11">
      <stop offset="0" stop-color="white"/>
      <stop offset="1" stop-color="black"/>
    </linearGradient>
    <rect id="b" width="1.23456" height="2.34567"/>
  </defs>
  <use href="#b"/>
  <use x="2" y="3" href="#b"/>
  <use x="4" y="6" href="#b"/>
  <rect x="6" y="9" width="1.23456" height="2.34567" fill="url(#a)"/>
</svg>
